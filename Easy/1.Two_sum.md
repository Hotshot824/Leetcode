### [1. Two sum]

``` go
func twoSum(nums []int, target int) []int {
	answer := []int{}
	for i := 0; i < len(nums); i++ {
		for j := 1; j < len(nums); j++ {
			if nums[i]+nums[j] == target && i != j {
				answer = append(answer, i)
				answer = append(answer, j)

				return answer
			}
		}
	}
	return nil
}
```
