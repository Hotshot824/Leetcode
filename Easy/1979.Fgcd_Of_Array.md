### 1979. Find Greatest Common Divisor of Array

```go
func findGCD(nums []int) int {
	var max, min int = nums[0], nums[0];
    for i := 0; i < len(nums); i++ {
		if nums[i] > max {
			max = nums[i];
		}
		if nums[i] < min {
			min = nums[i];
		}
	}
	return GCD(max, min);
}

func GCD(a int, b int) int {
	if b == 0 {
		return a;
	} else {
		return GCD(b, (a % b));
	}
}
```
