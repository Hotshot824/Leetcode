### [50. Pow(x, n)]

```go
func myPow(x float64, n int) float64 {
	if n >= 0 {
		return myPowImp(x, n)
	} else {
		return 1 / myPowImp(x, n)
	}
}

func myPowImp(x float64, n int) float64 {
	if n == 0 {
		return 1
	}
	if n%2 == 0 {
		return myPowImp(x*x, n/2) * 1
	} else {
		return myPowImp(x*x, n/2) * x
	}
}
```
