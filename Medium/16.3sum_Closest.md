### 16. 3Sum_Closest

```go 
func abs(num int) int {
	if num < 0 {
		num = -num
	}
	return num
}

func threeSumClosest(nums []int, target int) int {
	sort.Ints(nums)
	len := len(nums)
	AbsE := int(math.Pow10(4))
	for i := 0; i < len-2; i++ {
		var j, k int = i + 1, len - 1
		for j < k {
			diff := target - nums[i] - nums[j] - nums[k]
			if diff == 0 {
				return target
			}
			if abs(diff) < abs(AbsE) {
				AbsE = diff
			}
			if diff < 0 {
				k--
			} else {
				j++
			}
		}
	}
	return target - AbsE
}
```
