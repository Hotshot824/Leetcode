### 2. Add Two Numbers

/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */

func setLinkedlist(array []int, head *ListNode) {
	var current *ListNode
	current = head
	for i := 1; i < len(array); i++ {
		temp := ListNode{
			Val:  array[i],
			Next: nil,
		}
		current.Next = &temp
		current = current.Next
	}
}

func createNode() *ListNode {
	node := ListNode{
		Val:  0,
		Next: nil,
	}
	return &node
}

func checkCar(temp int) (val int, car int) {
	val, car = temp, 0
	if val > 9 {
		val = val % 10
		car = 1
	}
	return val, car
}

func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
	head := ListNode{
		Val:  0,
		Next: nil,
	}
	var current *ListNode
	var car, firstState int = 0, 0
	for l1 != nil && l2 != nil {

		if firstState == 0 {
			current = &head
			firstState++
		} else {
			current.Next = createNode()
			current = current.Next
		}

		temp := l1.Val + l2.Val + car
		car = 0
		temp, car = checkCar(temp)
		current.Val = temp

		l1 = l1.Next
		l2 = l2.Next
	}

	for l1 != nil {
		current.Next = createNode()
		current = current.Next
		temp := l1.Val + car
		car = 0
		temp, car = checkCar(temp)
		current.Val = temp

		l1 = l1.Next
	}

	for l2 != nil {
		current.Next = createNode()
		current = current.Next
		temp := l2.Val + car
		car = 0
		temp, car = checkCar(temp)
		current.Val = temp
		l2 = l2.Next
	}

	if car != 0 {
		current.Next = createNode()
		current = current.Next
		current.Val = car
	}

	return &head
}
