### 11. Container With Most Water

```go
func maxArea(height []int) int {
	len := len(height)
	var i, j int = 0, len - 1
	var maxArea, temp int = 0, 0

	for len > 0 {
		if height[i] > height[j] {
			temp = height[j] * (len - 1)
			j--
		} else {
			temp = height[i] * (len - 1)
			i++
		}

		if temp > maxArea {
			maxArea = temp
		}
		len--
	}
	return maxArea
}
```
